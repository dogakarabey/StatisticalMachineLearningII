# 2. Data Loading and Preprocessing

def load_and_preprocess_binary_data(hdf5_path, species1_name, species2_name,
                                    img_height, img_width,
                                    test_split_size=0.2, val_split_size=0.25):
    print(f"--- Loading BINARY data for {species1_name} vs {species2_name} ---")
    all_X_list = []
    all_y_list = []
    with h5py.File(hdf5_path, 'r') as f:
        s1_data = f[species1_name][:].astype(np.float32)
        print(f"  Raw {species1_name} samples: {s1_data.shape[0]}")
        for i in range(s1_data.shape[0]):
            spec = s1_data[i]
            if spec.shape[1] < img_width: spec = np.pad(spec, ((0, 0), (0, img_width - spec.shape[1])), mode='constant')
            elif spec.shape[1] > img_width: spec = spec[:, :img_width]
            if spec.shape[0] < img_height: spec = np.pad(spec, ((0, img_height - spec.shape[0]), (0,0)), mode='constant')
            elif spec.shape[0] > img_height: spec = spec[:img_height, :]
            all_X_list.append(spec)
            all_y_list.append(0)
        s2_data = f[species2_name][:].astype(np.float32)
        print(f"  Raw {species2_name} samples: {s2_data.shape[0]}")
        for i in range(s2_data.shape[0]):
            spec = s2_data[i]
            if spec.shape[1] < img_width: spec = np.pad(spec, ((0, 0), (0, img_width - spec.shape[1])), mode='constant')
            elif spec.shape[1] > img_width: spec = spec[:, :img_width]
            if spec.shape[0] < img_height: spec = np.pad(spec, ((0, img_height - spec.shape[0]), (0,0)), mode='constant')
            elif spec.shape[0] > img_height: spec = spec[:img_height, :]
            all_X_list.append(spec)
            all_y_list.append(1)
    X = np.array(all_X_list, dtype=np.float32)
    y = np.array(all_y_list, dtype=np.float32)
    print(f"Combined X shape: {X.shape}, Combined y shape: {y.shape}")
    X_train_val, X_test_orig, y_train_val, y_test = train_test_split(X, y, test_size=test_split_size, stratify=y, random_state=RANDOM_STATE_SPLIT)
    X_train_orig, X_val_orig, y_train, y_val = train_test_split(X_train_val, y_train_val, test_size=val_split_size, stratify=y_train_val, random_state=RANDOM_STATE_TRAIN_VAL_SPLIT)
    min_val, max_val, epsilon = np.min(X_train_orig), np.max(X_train_orig), np.float32(1e-7)
    X_train_norm, X_val_norm, X_test_norm = (X_train_orig - min_val) / (max_val - min_val + epsilon), (X_val_orig - min_val) / (max_val - min_val + epsilon), (X_test_orig - min_val) / (max_val - min_val + epsilon)
    X_train, X_val, X_test = np.expand_dims(X_train_norm, axis=-1), np.expand_dims(X_val_norm, axis=-1), np.expand_dims(X_test_norm, axis=-1)
    print(f"Final BINARY X_train: {X_train.shape}, y_train: {y_train.shape}\n")
    return (X_train, y_train), (X_val, y_val), (X_test, y_test)

def load_and_preprocess_multiclass_data(hdf5_path, img_height, img_width, num_classes,
                                        test_split_size=0.2, val_split_size=0.25):
    print(f"--- Loading MULTICLASS data ---")
    all_X_list = []
    all_y_labels_list = [] 

    label_encoder = LabelEncoder()

    with h5py.File(hdf5_path, 'r') as f:
        species_names = sorted(list(f.keys())) 
        print(f"Found species: {species_names}")
        label_encoder.fit(species_names)

        for species_idx, species_name in enumerate(species_names):
            s_data = f[species_name][:].astype(np.float32)
            print(f"  Raw {species_name} samples: {s_data.shape[0]}")
            for i in range(s_data.shape[0]):
                spec = s_data[i]
                if spec.shape[1] < img_width: spec = np.pad(spec, ((0, 0), (0, img_width - spec.shape[1])), mode='constant')
                elif spec.shape[1] > img_width: spec = spec[:, :img_width]
                if spec.shape[0] < img_height: spec = np.pad(spec, ((0, img_height - spec.shape[0]), (0,0)), mode='constant')
                elif spec.shape[0] > img_height: spec = spec[:img_height, :]
                all_X_list.append(spec)
                all_y_labels_list.append(species_name) 

    X = np.array(all_X_list, dtype=np.float32)
    y_integers = label_encoder.transform(all_y_labels_list)
    
    y = to_categorical(y_integers, num_classes=num_classes) 

    print(f"Combined X shape: {X.shape}, Combined y shape: {y.shape}")

    X_train_val, X_test_orig, y_train_val, y_test = train_test_split(X, y, test_size=test_split_size, stratify=y, random_state=RANDOM_STATE_SPLIT)
    X_train_orig, X_val_orig, y_train, y_val = train_test_split(X_train_val, y_train_val, test_size=val_split_size, stratify=y_train_val, random_state=RANDOM_STATE_TRAIN_VAL_SPLIT)

    min_val, max_val, epsilon = np.min(X_train_orig), np.max(X_train_orig), np.float32(1e-7)
    X_train_norm, X_val_norm, X_test_norm = (X_train_orig - min_val) / (max_val - min_val + epsilon), (X_val_orig - min_val) / (max_val - min_val + epsilon), (X_test_orig - min_val) / (max_val - min_val + epsilon)

    X_train, X_val, X_test = np.expand_dims(X_train_norm, axis=-1), np.expand_dims(X_val_norm, axis=-1), np.expand_dims(X_test_norm, axis=-1)

    print(f"Final MULTICLASS X_train: {X_train.shape}, y_train: {y_train.shape}\n")
    return (X_train, y_train), (X_val, y_val), (X_test, y_test), label_encoder.classes_


# 3. Binary Model
def create_simple_binary_cnn(input_shape, learning_rate=0.001):
    model = Sequential([
        Conv2D(16, (3, 3), activation='relu', input_shape=input_shape, padding='same'),
        MaxPooling2D((2, 2)),
        Dropout(0.25),
        Conv2D(32, (3, 3), activation='relu', padding='same'),
        MaxPooling2D((2, 2)),
        Dropout(0.25),
        Flatten(),
        Dense(64, activation='relu'),
        Dropout(0.5),
        Dense(1, activation='sigmoid')
    ])
    optimizer = Adam(learning_rate=learning_rate)
    model.compile(optimizer=optimizer, loss='binary_crossentropy', metrics=['accuracy'])
    return model

def create_simple_multiclass_cnn(input_shape, num_classes, learning_rate=0.001):
    model = Sequential([
        Conv2D(32, (3, 3), activation='relu', input_shape=input_shape, padding='same'),
        MaxPooling2D((2, 2)),
        Dropout(0.2),

        Conv2D(64, (3, 3), activation='relu', padding='same'),
        MaxPooling2D((2, 2)),
        Dropout(0.3),

        Conv2D(128, (3, 3), activation='relu', padding='same'),
        MaxPooling2D((2, 2)),
        Dropout(0.4),

        GlobalAveragePooling2D(),

        Dense(128, activation='relu'),
        Dropout(0.5),
        Dense(num_classes, activation='softmax')
    ])
    optimizer = Adam(learning_rate=learning_rate)
    model.compile(optimizer=optimizer, loss='categorical_crossentropy', metrics=['accuracy'])
    return model

#4. Plotting
def plot_history(history, model_name="Model"):
    plt.figure(figsize=(12, 4))
    plt.subplot(1, 2, 1)
    plt.plot(history.history['accuracy'], label='Train Acc')
    plt.plot(history.history['val_accuracy'], label='Val Acc')
    plt.title(f'{model_name} Accuracy')
    plt.ylabel('Accuracy')
    plt.xlabel('Epoch')
    plt.legend(loc='upper left')
    plt.subplot(1, 2, 2)
    plt.plot(history.history['loss'], label='Train Loss')
    plt.plot(history.history['val_loss'], label='Val Loss')
    plt.title(f'{model_name} Loss')
    plt.ylabel('Loss')
    plt.xlabel('Epoch')
    plt.legend(loc='upper left')
    plt.tight_layout()
    plt.show()
